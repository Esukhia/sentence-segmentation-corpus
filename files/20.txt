ལྷ་ ལ་ བཟང་ རྒྱུ་ མེད་ པར་ ཤེས་ ཏེ། དངོས་ གྲུབ་ རང་ ལས་ བྱུང་ བ་ དང་ ། བདག་ ལ་ ངན་ རྒྱུ་ མེད་ པར་ རྟོགས་ པས། དེ་ ཉིད་ སྒྱུ་ ལུས་ ལྷའི་ སྐུར་ གསལ་ བར་ ནུས་ པས་ འོག་ མ་ ལས་ ཁྱད་ པར་ དུ་ འཕགས་ ཤིང་ ། བསྐྱེད་ རྒྱུའི་ ལྷ་ དང་ བདག་ གཉིས་ སུ་ མེད་ པར་ རྟོགས་ པས། དོན་ དམ་ ཀུན་ རྫོབ་ ཀྱི་ མིང་ མེད་ ཅིང་ ཟབ་ མོ་ རྣལ་ འབྱོར་ གྱི་ ཆ་ ལ་ སྙོམས་ པར་ སྤྱོད་ ཅིང་ ། ཕྱག་ རྒྱ་ བཞི་ པ་ ཏིང་ ངེ་ འཛིན་ གྱི་ སྒོ་ ནས་ བདེན་ པ་ དབྱེར་ མེད་ ཟབ་ གསལ་ གཉིས་ སུ་ མེད་ པའི་ ཡེ་ ཤེས་ སྒོམ་ ཞིང་ ལང་ ཚོས་ སྙིང་ ལ་ འགྲོགས་ པས་ ལག་ པ་ ཟུང་ འབྲེལ་ བདེ་ བས་ ཚིམ་ པའོ། །ཞེས་ སྔགས་ ཕྱི་ པའི་ རྣམ་ གཞག་ འགག་ ལ་ འདྲིལ་ བར་ གསུངས་ ཤིང་ ། དེ་ ལྟར་ སྔགས་ ཕྱི་ ནང་ གི་ ཁྱད་ པར་ གང་ གིས་ འབྱེད་ ཅེ་ ན། ལྟ་ སྤྱོད་ ཏིང་ ངེ་ འཛིན་ གསུམ་ དབྱེར་ མེད་ དུ་ འདོད་ པ་ སྔགས་ ནང་ པ། མི་ འདོད་ པ་ སྔགས་ ཕྱི་ པའོ། །ཁ་ ཅིག་ ན་ རེ། དེ་ ལྟ་ ན་ སྔགས་ ཕྱི་ པ་ དང་ མཚན་ ཉིད་ ཀྱི་ ཐེག་ པ་ ལ་ ཁྱད་ པར་ མེད་ དམ་ སྙམ་ ན། ཡོད་ དེ་ མཚན་ ཉིད་ ཐེག་ པས་ ནི་ ཀུན་ རྫོབ་ ལམ་ དུ་ བྱེད་ མི་ ནུས་ ལ། གསང་ སྔགས་ ཕྱི་ པས་ ནི་ ཀུན་ རྫོབ་ ལམ་ དུ་ བྱེད་ ནུས་ པས་ སོ། །ད་ ནི་ ལྟ་ སྤྱོད་ ཏིང་ ངེ་ འཛིན་ གསུམ་ གྱི་ སྒོ་ ནས་ དབྱེ་ བ་ ནང་ རྒྱུད་ གསུམ་ གྱི་ རྣམ་ གཞག་ བརྗོད་ དེ།

ཀྱི་ ཐེག་ པ། སེམས་ ཉིད་ སངས་ རྒྱས་ ཀྱི་ འབྲས་ བུར་ འདོད་ ཅིང་ ཀུན་ རྫོབ་ མི་ སྤོང་ བར་ ལམ་ དུ་ འཁྱེར་ བ་ ནི་ འབྲས་ བུ་ གསང་ སྔགས་ ཀྱི་ ཐེག་ པ་ ཞེས་ འདོད་ དོ། །ཐེག་ པ་ དེ་ དག་ ཐམས་ ཅད་ ཀྱི་ མཐའ་ བརྟེན་ གྱི་ འབྲས་ བུ་ ནི། གཏེར་ ཆེན་ ཐམས་ ཅད་ མཁྱེན་ པས། རྗོད་ བྱེད་ སྡེ་ སྣོད་ གསུམ་ གྱི་ བརྗོད་ བྱ་ རྣམ་ དག་ ཚུལ་ ཁྲིམས་ ཀྱི་ བསླབ་ པས་ འདུལ་ བ། མི་ གཡོ་ ཏིང་ འཛིན་ གྱི་ བསླབ་ པས་ མདོ་ སྡེ། སོ་ སོར་ རྟོག་ པའི་ ཤེས་ རབ་ ཀྱི་ བསླབ་ པས་ མངོན་ པ། རྒྱུད་ སྡེ་ འོག་ མ་ གསུམ་ ལས་ བྱ་ སྤྱོད་ ཀྱི་ ལུས་ ངག་ ཡིད་ གསུམ་ ལྷ་ སྔགས་ ཏིང་ ངེ་ འཛིན་ གསུམ་ དུ་ གསལ་ བས་ མཁའ་ སྤྱོད་ ལ་ སོགས་ པའི་ ཐུན་ མོང་ གི་ དངོས་ གྲུབ་ བརྒྱད་ རྡོ་ རྗེ་ གུར་ ལས། མིག་ སྨན་ དང་ ནི་ རྐང་ མགྱོགས་ དང་ ། །རལ་ གྲི་ དང་ ནི་ ས་ འོག་ འགྲུབ། །རིལ་ བུ་ དང་ ནི་ མཁའ་ སྤྱོད་ ཉིད། །མི་ སྣང་ བ་ དང་ བཅུད་ ཀྱི་ ལེན། །ཞེས་ གསུངས་ པ་ ལྟར་ རོ། །དང་ ། ཡོ་ གས་ མཆོག་ གི་ དངོས་ གྲུབ་ བསྒྲུབས་ ཏེ། སྐྱེ་ བ་ བཅུ་ དྲུག་ ཚུན་ དུ་ འབྲས་ བུ་ ཐོབ་ ཏུ་ འདོད་ པ་ དང་ །

མ་ ཧཱ་ ཨ་ ནུ་ གཉིས་ ཀྱིས་ རྡོ་ རྗེ་ ལུས་ ལ་ རྩ་ ཐིག་ རླུང་ གསུམ་ གྱིས་ གནད་ དུ་ བསྣུན་ པས།

སྐྱེ་ བ་ གསུམ་ མམ་ བདུན་ ལས་ མི་ འདའ་ བའི་ འབྲས་ བུ་ མངོན་ བྱེད་ དང་ ། ཁྱད་ པར་ མཆོག་ གི་ ཨ་ ཏིས་ སྣང་ བ་ བཞི་ གཏན་ ལ་ ཕབ་ པས་ སྐྱེ་ ཤི་ བར་ དོ་ གསུམ་ གང་ རུང་ ལ་ ངེས་ པར་ གྲོལ་ ཞེས་ གསུངས་ སོ། །རྙིང་ མ་ བཀའ་ ལུགས་ ལ་ འདུས་ པ་ མདོ་ དབང་ ཞེས་ ཐེག་ པ་ རིམ་ པ་ དགུའི་ དབང་ བསྐུར་ བ་ ཡང་ འདི་ ཉིད་ ཡིན་ ཞིང་ ། འོན་ ཀྱང་ རྒྱུའི་ ཐེག་ པ་ ལ་ དབང་ བསྐུར་ གྱི་ ཐ་ སྙད་ གྲགས་ པ་ ལུགས་ སྲོལ་ འདིའི་ ཁྱད་ ཆོས་ ཡིན་ གཤིས་ སྔ་ རབས་ པ་ དག་ གིས་ དགག་ པ་ སྣ་ ཚོགས་ བྱུང་ མོད་ དེ། བརྟག་ གཉིས་ ལས། དང་ པོར་ གསོ་ སྦྱོང་ སྦྱིན་ པར་ བྱ། །དེ་ རྗེས་ བྱེ་ བྲག་ སྨྲ་ བ་ བསྟན། །མདོ་ སྡེ་ པ་ ཡང་ དེ་ བཞིན་ ནོ༑ ༑ཞེས་ སོགས་ ཀྱི་ མཐར། དེ་ རྗེས་ ཀྱཻ་ ཡི་ རྡོ་ རྗེ་ བསྟན། །ཞེས་ ཐེག་ པ་ རིམ་ པར་ འཇུག་ པ་ དང་ མཐུན་ པར་ འདིར་ སྔགས་ ལམ་ དུ་ ཐེག་ པ་ ཀུན་ གྱི་ གཞུང་ ལམ་ ཚང་ བར་ བསྟན་ པར་ མ་ ཟད་ དབང་ བསྐུར་ གྱི་ བརྗོད་ བྱའི་ དོན་ ལ་ མི་ འཐད་ པ་ འགའ་ ཡང་ མེད་ ལ། རྒྱུ་ འབྲས་ ཀྱི་ ཐེག་ པ་ གཉིས་ ལྟ་ སྤྱོད་ འདྲེས་ པའི་ སྐྱོན་ ཡང་ མི་ མངའ། དབང་ གི་ དོན་ ནི་ ལྷག་ གནས་ སམ་ བྱིན་ གྱིས་ བརླབ་ པ་ ཡིན་ པས། བྱིན་ བརླབ་ ནི་ འདུལ་ བའི་ ལས་ ཆོག་ ལས་ ཀྱང་ ། བདག་ མིང་ འདི་ ཞེས་ བགྱི་ བའི་ ཆོས་ གོས་ འདི་ སྣམ་ སྦྱར་ དུ་ བྱིན་ གྱིས་ བརླབས་ སོ། །ཞེས་ པས་ མཚོན་ ཏེ་ ཤིན་ ཏུ་ མང་ ། ཁྱད་ པར་ དངོས་ བསྟན་ ནི༑ སུམ་ བརྒྱ་ པ་ ལས། ངེས་ པར་ འབྱུང་ བའི་ ཚུལ་ ཁྲིམས་ སྡུག་ བསྔལ་ སྤོང་ ། །

ཤིན་ ཏུ་ རྣལ་ འབྱོར་ ཞེས་ ཨ་ ཏི་ ཡོ་ གའི་ རྒྱུད་ སྡེ་ བསྣན་ པས་ དྲུག་ ཏུ་ བྱུང་ བ་ ཡིན་ ལ་ རྒྱུད་ སྡེའི་ རྣམ་ གྲངས་ ཙམ་ ལ་ འཕགས་ ཡུལ་ གྱི་ སློབ་ དཔོན་ ཁ་ ཅིག་ བདུན་ དང་ ལྔར་ བཞེད་ པ་ ཡང་ གསུངས་ ལ། ཨ་ ནུ་ ཨ་ ཏི་ གཉིས་ པོ་ ཐེག་ པའི་ རྐང་ གྲངས་ སུ་ འདྲེན་ པ་ སྔགས་ རྙིང་ མའི་ རང་ སྐད་ ལྟ་ བུར་ འདུག་ པ་ ལ་ མི་ རིགས་ སོ། །ཞེས་ བརྒལ་ བ་ ཡང་ ཧ་ ཅང་ ཐལ་ ཏེ། སྤྱིར་ ཐེག་ པའི་ གོ་ བ་ ནི། བཞེད་ ཚུལ་ ཐ་ དད་ ཀྱང་ འོག་ ནས་ གོང་ དུ་ འདེགས་ པར་ བྱེད་ པས་ ན་ ཐེག་ པ་ ཞེས་ ཡོན་ ཏན་ རིན་ པོ་ ཆེ་ སྡུད་ པ་ ལས་ གསུངས་ པ་ ལྟར་ ཡིན་ ལ། དེ་ ཡང་ འདི་ ལྟ་ སྟེ། དང་ པོ་ ཕྱི་ མཚན་ ཉིད་ ཀྱི་ ཐེག་ པ་ གསུམ་ པོས་ རིམ་ བཞིན་ དགྲ་ བཅོམ་ པ་ དང་ ། ནེ་ ཙོའམ་ བསེ་ རུ་ ལྟ་ བུའི་ རང་ སངས་ རྒྱས་ དང་ ། བཅུ་ གཅིག་ ཀུན་ ཏུ་ འོད་ ཀྱི་ ས་ མཐར་ འདེགས་ པར་ བྱེད་ པས་ ན་ ཐེག་ པ་ དང་ ། གཉིས་ པ་ ཕྱི་ ཐུབ་ པ་ རྒྱུད་ ཀྱི་ ཐེག་ པ་ གསུམ་ པོས་ རིམ་ པ་ བཞིན་ རིགས་ གསུམ་ རྡོ་ རྗེ་ འཛིན་ པའི་ ས་ དང་ །

པའི་ གདུལ་ བྱ་ རྣམས་ རྗེས་ སུ་ གཟུང་ བ་ དང་ ། གཟུ་ བོའི་ ཡིད་ ཅན་ རྣམས་ མགུ་ བ་ བསྐྱེད་ པའི་ ཕྱིར་ སྔ་ འགྱུར་ ཡོ་ ག་ གསུམ་ གྱི་ རྒྱུད་ ཚུལ་ ཐ་ དད་ དུ་ ཕྱེ་ ནས་ བཤད་ པ་ དང་ ། རྣམ་ པ་ ཐམས་ ཅད་ མཁྱེན་ པའི་ ཡེ་ ཤེས་ ལ་ བག་ ཆགས་ གཞག་ པའི་ སླད་ དུ་ ཐེག་ རྩེ་ ཤིན་ ཏུ་ རྣལ་ འབྱོར་ གྱི་ གཞི་ ལམ་ འབྲས་ བུའི་ རྣམ་ གཞག་ མདོར་ བསྟན་ ནས་ བཤད་ པའོ། །དང་ པོ་ ནི། དམ་ ཆོས་ པད་ དཀར་ ལས། བསྐལ་ པ་ བྱེ་ སྟོང་ བསམ་
